List

public class IntList{
    public int first;
    public IntList rest;

    public IntList(int f, IntList r){
        first = f;
        rest = r;
    }

    public int size(){
        if (rest == null)
            return 1;
        return 1 + this.rest.size()
        }

    public static void main(String args[]){
        IntList L = new IntList();
        L = new IntList(10, L);
        
    }
}
