yield from a

def substrings(s):

Example: partitions

def partitions(n, m):
    if n>0 and m>0:
        if n==m:
            yield str(m)
        for p in partiotions(n-m, m):
            yield p+'+'+str(m)
        yield from partitions(n, m-1)
